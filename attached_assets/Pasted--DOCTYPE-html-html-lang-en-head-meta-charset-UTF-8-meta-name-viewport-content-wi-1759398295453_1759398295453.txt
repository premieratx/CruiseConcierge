    const SUPABASE_URL = 'https://tgambsdjfwgoohkqopns.supabase.co';
    const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InRnYW1ic2RqZndnb29oa3FvcG5zIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NTkzNDYzMDUsImV4cCI6MjA3NDkyMjMwNX0.xRGHgSXJsMkxO5KV-Uh7TvLPGd8MnbYrBdKi-QNUMh4';
    const STRIPE_KEY = 'pk_live_51M4E7yHwOvgmegQ30ldL6EdaHzDcYJjHhyir1EPDzAQ7Wf846XECS93ouhqZp5itCYmfNzcoU0L8LrKPY9vqNTEZ00ylzHoxte';
    
    let state = {
      step: 1,
      selectedDate: null,
      partyType: null,
      guestCount: null,
      leadId: null,
      customerId: null,
      discoSlots: [],
      privateSlots: []
    };

    function showStep(step) {
      document.querySelectorAll('.step').forEach(el => el.classList.add('hidden'));
      document.getElementById('step' + step).classList.remove('hidden');
      state.step = step;
      
      // Post height to parent for iframe resize
      setTimeout(() => {
        const height = document.body.scrollHeight;
        window.parent.postMessage({ type: 'quote-widget-height', height }, '*');
      }, 100);
    }

    function selectDate() {
      const date = document.getElementById('dateInput').value;
      if (!date) {
        alert('Please select a date');
        return;
      }
      state.selectedDate = date;
      showStep(2);
    }

    function selectPartyType(type) {
      state.partyType = type;
      showStep(3);
    }

    function selectGuestCount() {
      const count = parseInt(document.getElementById('guestInput').value);
      if (!count || count < 1) {
        alert('Please enter number of guests');
        return;
      }
      state.guestCount = count;
      
      // Update summary
      document.getElementById('summaryDate').textContent = state.selectedDate;
      document.getElementById('summaryParty').textContent = state.partyType;
      document.getElementById('summaryGuests').textContent = state.guestCount;
      
      showStep(4);
    }

    function goBack(toStep) {
      showStep(toStep);
    }

    function editDate() {
      showStep(1);
    }

    async function captureLead() {
      const firstName = document.getElementById('firstName').value;
      const lastName = document.getElementById('lastName').value;
      const email = document.getElementById('email').value;
      const phone = document.getElementById('phone').value;

      if (!firstName || !lastName || !email || !phone) {
        alert('Please fill in all fields');
        return;
      }

      try {
        // Create lead
        const response = await fetch(SUPABASE_URL + '/functions/v1/create-lead', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'apikey': SUPABASE_KEY
          },
          body: JSON.stringify({
            first_name: firstName,
            last_name: lastName,
            email,
            phone,
            event_date: state.selectedDate,
            party_type: state.partyType,
            guest_count: state.guestCount,
            quote_url: window.location.href
          })
        });

        const data = await response.json();
        
        if (data.lead_id) {
          state.leadId = data.lead_id;
          state.customerId = data.customer_id;
          
          // Update summary
          document.getElementById('summaryDate2').textContent = state.selectedDate;
          document.getElementById('summaryParty2').textContent = state.partyType;
          document.getElementById('summaryGuests2').textContent = state.guestCount;
          
          showStep(5);
          fetchAvailableSlots();
        }
      } catch (error) {
        console.error('Error creating lead:', error);
        alert('Something went wrong. Please try again.');
      }
    }

    async function fetchAvailableSlots() {
      document.getElementById('loadingSlots').classList.remove('hidden');
      document.getElementById('slotsContainer').classList.add('hidden');

      try {
        const response = await fetch(SUPABASE_URL + '/functions/v1/list-time-slots', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'apikey': SUPABASE_KEY
          },
          body: JSON.stringify({
            date: state.selectedDate,
            party_type: state.partyType,
            guest_count: state.guestCount
          })
        });

        const data = await response.json();
        
        state.discoSlots = data.discoSlots || [];
        state.privateSlots = data.privateSlots || [];

        displaySlots();
      } catch (error) {
        console.error('Error fetching slots:', error);
        document.getElementById('loadingSlots').innerHTML = '<p style="color: #ef4444;">Error loading cruises. Please try again.</p>';
      }
    }

    function displaySlots() {
      document.getElementById('loadingSlots').classList.add('hidden');
      const container = document.getElementById('slotsContainer');
      container.classList.remove('hidden');
      container.innerHTML = '';

      const isBach = state.partyType === 'bachelor' || state.partyType === 'bachelorette';

      if (!isBach && state.discoSlots.length > 0) {
        const discoSection = document.createElement('div');
        discoSection.innerHTML = '<h3 style="font-size: 20px; font-weight: bold; margin-bottom: 16px;">Disco Cruises</h3>';
        
        state.discoSlots.forEach(slot => {
          const card = createSlotCard(slot, 'disco');
          discoSection.appendChild(card);
        });
        
        container.appendChild(discoSection);
      }

      if (state.privateSlots.length > 0) {
        const privateSection = document.createElement('div');
        privateSection.style.marginTop = '32px';
        privateSection.innerHTML = '<h3 style="font-size: 20px; font-weight: bold; margin-bottom: 16px;">Private Cruises</h3>';
        
        state.privateSlots.forEach(slot => {
          const card = createSlotCard(slot, 'private');
          privateSection.appendChild(card);
        });
        
        container.appendChild(privateSection);
      }

      if (state.discoSlots.length === 0 && state.privateSlots.length === 0) {
        container.innerHTML = '<div class="card"><p>No available cruises found for your date. Please try a different date.</p></div>';
      }

      // Resize iframe
      setTimeout(() => {
        const height = document.body.scrollHeight;
        window.parent.postMessage({ type: 'quote-widget-height', height }, '*');
      }, 100);
    }

    function createSlotCard(slot, type) {
      const card = document.createElement('div');
      card.className = 'cruise-option';
      
      const startTime = new Date(slot.start_at).toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' });
      const endTime = new Date(slot.end_at).toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' });
      const price = parseFloat(slot.price);

      card.innerHTML = `
        <div style="display: flex; justify-content: space-between; align-items: start;">
          <div>
            <h4 style="font-weight: 600; font-size: 18px; margin-bottom: 8px;">${type === 'disco' ? 'Disco Cruise' : 'Private Cruise'}</h4>
            <p style="color: #6b7280; margin-bottom: 4px;">${startTime} - ${endTime}</p>
            <p style="color: #6b7280;">Capacity: ${slot.capacity_available} available</p>
          </div>
          <div style="text-align: right;">
            <p style="font-size: 24px; font-weight: bold; color: #3b82f6;">$${price.toFixed(2)}</p>
            <button onclick="bookSlot('${slot.id}', ${price})" class="btn btn-primary" style="margin-top: 8px;">Book Now</button>
          </div>
        </div>
      `;

      return card;
    }

    async function bookSlot(slotId, price) {
      try {
        const response = await fetch(SUPABASE_URL + '/functions/v1/create-stripe-checkout', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'apikey': SUPABASE_KEY
          },
          body: JSON.stringify({
            time_slot_id: slotId,
            customer_id: state.customerId,
            package_type: 'standard',
            headcount: state.guestCount,
            party_type: state.partyType,
            amount: price,
            deposit_amount: price * 0.5,
            success_url: window.location.origin + '/booking-success',
            cancel_url: window.location.href
          })
        });

        const data = await response.json();
        
        if (data.sessionId) {
          const stripe = Stripe(STRIPE_KEY);
          await stripe.redirectToCheckout({ sessionId: data.sessionId });
        }
      } catch (error) {
        console.error('Error creating checkout:', error);
        alert('Something went wrong. Please try again.');
      }
    }

    document.getElementById('dateInput').min = new Date().toISOString().split('T')[0];
