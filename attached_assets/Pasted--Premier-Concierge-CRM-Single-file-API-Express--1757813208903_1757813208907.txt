// Utilities
const genId = (p) => `${p}_${crypto.randomBytes(6).toString("hex")}`;
const nowIso = () => new Date().toISOString();
const int = (x) => Math.round(Number(x) || 0);
const inDays = (d) => d * 24 * 3600 * 1000;

// Simple tagging helpers
function addTagToEntity(arr, id, tag) {
  const ent = arr.find((e) => e.id === id);
  if (!ent) return;
  ent.tags = ent.tags || [];
  if (!ent.tags.includes(tag)) ent.tags.push(tag);
}

// Pricing math (integer cents)
function applyDiscounts({ orgId, items, promoCode, projectDate }) {
  const orgRules = db.discountRules.filter((r) => r.orgId === orgId && r.auto);
  const activeAuto = orgRules.filter((r) => {
    const now = Date.now();
    const s = r.startDate ? new Date(r.startDate).getTime() : null;
    const e = r.endDate ? new Date(r.endDate).getTime() : null;
    if (s && now < s) return false;
    if (e && now > e) return false;
    return true;
  });
  let subtotal = items.reduce((s, it) => s + int(it.unitPrice) * int(it.qty), 0);
  let tax = 0; // add your tax rules here if needed
  let discounts = [];
  // promo
  if (promoCode) {
    const promo = db.promos.find((p) => p.code?.toUpperCase() === promoCode.toUpperCase() && p.active !== false);
    if (promo) {
      const within =
        (!promo.startsAt || Date.now() >= new Date(promo.startsAt).getTime()) &&
        (!promo.endsAt || Date.now() <= new Date(promo.endsAt).getTime());
      if (within) {
        let amt = 0;
        if (promo.percentBasis) amt = Math.floor((subtotal * promo.percentBasis) / 10000);
        if (promo.amountCents) amt = Math.min(subtotal, (amt || 0) + promo.amountCents);
        discounts.push({ source: "PROMO", code: promo.code, amount: amt });
      }
    }
  }
  // auto rules (priority asc, stackable logic)
  const sorted = activeAuto.sort((a, b) => (a.priority || 100) - (b.priority || 100));
  let runningSubtotal = subtotal - discounts.reduce((s, d) => s + d.amount, 0);
  for (const r of sorted) {
    if (r.minSubtotal && runningSubtotal < r.minSubtotal) continue;
    let amt = 0;
    if (r.kind === "PERCENT") amt = Math.floor((runningSubtotal * (r.percentBasis || 0)) / 10000);
    if (r.kind === "FIXED") amt = Math.min(runningSubtotal, r.amountCents || 0);
    if (r.maxDiscountCents) amt = Math.min(amt, r.maxDiscountCents);
    if (amt > 0) {
      discounts.push({ source: "AUTO_RULE", ruleId: r.id, amount: amt });
      runningSubtotal -= amt;
      if (r.stackable === false) break;
    }
  }
  const discountTotal = discounts.reduce((s, d) => s + d.amount, 0);
  const total = Math.max(0, subtotal + tax - discountTotal);
  return { subtotal, discountTotal, tax, total, breakdown: discounts };
}

// Deposit policy
function evaluatePricingRules(orgId, projectDateIso) {
  const rules = db.pricingRules
    .filter((r) => r.orgId === orgId && r.active)
    .sort((a, b) => (a.priority || 100) - (b.priority || 100));
  let policy = { depositType: "DEPOSIT_25", depositDuePercent: 25 }; // default
  if (!projectDateIso) return policy;
  const now = new Date();
  const pdt = new Date(projectDateIso);
  const daysDiff = Math.ceil((pdt.getTime() - now.getTime()) / (24 * 3600 * 1000));
  for (const r of rules) {
    const within = r.condition?.withinDaysOfProjectDate;
    if (typeof within === "number" && daysDiff <= within) {
      if (r.action?.depositPolicy === "FULL_ON_ACCEPTANCE") {
        policy = { depositType: "FULL_ON_ACCEPTANCE", depositDuePercent: 100 };
        break;
      }
    }
  }
  return policy;
}

// HMAC token for quote open pixel
function signToken(payload) {
  const data = Buffer.from(JSON.stringify(payload)).toString("base64url");
  const sig = crypto.createHmac("sha256", APP_SECRET).update(data).digest("base64url");
  return `${data}.${sig}`;
}
function verifyToken(token) {
  const [data, sig] = token.split(".");
  const check = crypto.createHmac("sha256", APP_SECRET).update(data).digest("base64url");
  if (sig !== check) return null;
  try { return JSON.parse(Buffer.from(data, "base64url").toString("utf8")); } catch { return null; }
}

// accept quote → generate invoice + deposit policy
app.post("/quotes/:id/accept", async (req, res) => {
  const q = db.quotes.find((x) => x.id === req.params.id);
  if (!q) return res.status(404).json({ error: "not found" });
  const project = db.projects.find((p) => p.id === q.projectId);
  if (!project) return res.status(400).json({ error: "project missing" });
  const policy = evaluatePricingRules(q.orgId, project.projectDate);
  q.status = "ACCEPTED";
  addTagToEntity(db.projects, project.id, "quote-accepted");
  project.pipelinePhase = "ph_quote_accepted";
  // Create invoice
  const invId = genId("inv");
  const subtotal = q.subtotal;
  const tax = q.tax;
  const total = q.total;
  let schedule = [];
  if (policy.depositDuePercent >= 100) {
    schedule.push({ line: 1, due: "ON_ACCEPTANCE", percent: 100 });
  } else {
    schedule.push({ line: 1, due: "ON_ACCEPTANCE", percent: 25 });
    schedule.push({ line: 2, due: "BEFORE_PROJECT_DATE", percent: 75, daysBefore: 7 });
  }
  const balance = total;
  db.invoices.push({ id: invId, orgId: q.orgId, projectId: project.id, quoteId: q.id, status: "OPEN", subtotal, tax, total, balance, schedule, createdAt: nowIso() });
  // Create checkout session for the first due (percent)
  const firstPercent = schedule[0].percent;
  const amountCents = Math.floor((total * firstPercent) / 100);
  const session = await StripeShim.createCheckoutSession({
    invoiceId: invId,
    amountCents,
    successUrl: `${BASE_URL}/public/thanks?invoiceId=${invId}`,
    cancelUrl: `${BASE_URL}/public/cancel?invoiceId=${invId}`,
  });
  res.json({ invoiceId: invId, checkoutUrl: session.url });
});

// ---- Stripe webhook (deposit/remaining)
app.post("/webhooks/stripe", (req, res) => {
  const evt = StripeShim.verifyWebhook(req);
  try {
    if (evt.type === "checkout.session.completed" || evt.type === "payment_intent.succeeded") {
      const { invoiceId, amount } = evt.data || {};
      const inv = db.invoices.find((x) => x.id === invoiceId);
      if (inv) {
        const payId = genId("pay");
        db.payments.push({ id: payId, invoiceId, amount: int(amount), status: "SUCCEEDED", paidAt: nowIso(), method: "card" });
        inv.balance = Math.max(0, inv.balance - int(amount));
        addTagToEntity(db.projects, inv.projectId, inv.balance === 0 ? "payment-received" : "deposit-paid");
        const project = db.projects.find((p) => p.id === inv.projectId);
        if (project) {
          project.pipelinePhase = inv.balance === 0 ? "ph_paid" : "ph_deposit_paid";
        }
      }
    }
  } catch (e) {
    console.error("webhook error", e);
  }
  res.json({ received: true });
});

// ---- Leads vs Clients lists & tags
app.get("/contacts/leads", (req, res) => {
  const leads = db.contacts.filter((c) => {
    const projects = db.projects.filter((p) => p.contactId === c.id);
    const hasClientProject = projects.some((p) => ["ph_deposit_paid", "ph_paid"].includes(p.pipelinePhase));
    return !hasClientProject;
  });
  res.json(filterByQueryTagsSort(leads, req));
});
app.get("/contacts/clients", (req, res) => {
  const clients = db.contacts.filter((c) => {
    const projects = db.projects.filter((p) => p.contactId === c.id);
    const hasClientProject = projects.some((p) => ["ph_deposit_paid", "ph_paid"].includes(p.pipelinePhase));
    return hasClientProject;
  });
  res.json(filterByQueryTagsSort(clients, req));
});
function filterByQueryTagsSort(list, req) {
  const q = (req.query.query || "").toString().toLowerCase();
  const tags = ((req.query.tags || "") + "").split(",").map((t) => t.trim()).filter(Boolean);
  let out = [...list];
  if (q) out = out.filter((c) => (c.name || "").toLowerCase().includes(q) || (c.email || "").toLowerCase().includes(q));
  if (tags.length) out = out.filter((c) => (c.tags || []).some((t) => tags.includes(t)));
  const sort = (req.query.sort || "createdAt").toString();
  out.sort((a, b) => ((a[sort] || "") > (b[sort] || "")) ? 1 : -1);
  return out;
}

// ---- Chatbot action endpoint (safe, surface-level API for your AI bot)
app.post("/chatbot/action", (req, res) => {
  /** Expected payload examples:
   *  { type: "create_contact", data: { name, email, phone, tags } }
   *  { type: "draft_quote", data: { projectId, items[], promoCode } }
   *  { type: "send_quote", data: { quoteId, to, message } }
   *  { type: "move_pipeline", data: { projectId, phaseId } }
   */
  const { type, data } = req.body || {};
  try {
    switch (type) {
      case "create_contact": {
        const id = genId("ct");
        const { orgId = "org_demo", name, email, phone, tags = [] } = data || {};
        db.contacts.push({ id, orgId, name, email, phone, tags, createdAt: nowIso() });
        return res.json({ ok: true, id });
      }
      case "draft_quote": {
        const { projectId, items = [], promoCode = null } = data || {};
        const id = genId("qt");
        const q = { id, orgId: "org_demo", projectId, status: "DRAFT", items, promoCode, discountTotal: 0, breakdown: [], subtotal: 0, tax: 0, total: 0, version: 1, createdAt: nowIso() };
        // recompute totals
        const project = db.projects.find((p) => p.id === projectId);
        const price = applyDiscounts({
          orgId: "org_demo",
          items: items.map((it) => ({ unitPrice: int(it.unitPrice), qty: int(it.qty || 1) })),
          promoCode,
          projectDate: project?.projectDate || null,
        });
        Object.assign(q, { discountTotal: price.discountTotal, breakdown: price.breakdown, subtotal: price.subtotal, tax: price.tax, total: price.total });
        db.quotes.push(q);
        return res.json({ ok: true, id });
      }
      case "send_quote": {
        // delegate to the normal endpoint would be ideal; inline simple version here
        const q = db.quotes.find((x) => x.id === data.quoteId);
        if (!q) return res.status(404).json({ error: "quote not found" });
        const project = db.projects.find((p) => p.id === q.projectId);
        const contact = db.contacts.find((c) => c.id === project?.contactId);
        const to = data.to || contact?.email;
        if (!to) return res.status(400).json({ error: "no recipient email" });
        const token = signToken({ quoteId: q.id, contactId: contact?.id, projectId: project?.id, ts: Date.now() });
        const viewUrl = `${BASE_URL}/public/quotes/${q.id}`;
        const pixelUrl = `${BASE_URL}/quotes/track/open/${token}`;
        const html = `<p>Your quote is ready.</p><p><a href="${viewUrl}">View Quote</a></p><img src="${pixelUrl}" width="1" height="1" alt=""/>`;
        Email.send({ to, subject: "Your Quote", html });
        db.emails.push({ id: genId("em"), orgId: q.orgId, projectId: project?.id, to, subject: "Your Quote", body: html, sentAt: nowIso(), provider: "demo" });
        q.status = "SENT";
        project.pipelinePhase = "ph_quote_sent";
        addTagToEntity(db.projects, project.id, "quote-sent");
        return res.json({ ok: true });
      }
      case "move_pipeline": {
        const pr = db.projects.find((p) => p.id === data.projectId);
        if (!pr) return res.status(404).json({ error: "project not found" });
        pr.pipelinePhase = data.phaseId;
        return res.json({ ok: true });
      }
      default:
        return res.status(400).json({ error: "unsupported action" });
    }
  } catch (e) {
    console.error("chatbot action error", e);
    return res.status(500).json({ error: "internal" });
  }
});
